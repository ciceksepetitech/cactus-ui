# useIsCSR

## Importing useIsCSR Hook

```jsx
import { useIsCSR } from '@ciceksepeti/cui-hooks';
```

## Example

```jsx
import React from 'react';
import { useIsCSR } from '@ciceksepeti/cui-hooks';

export const Example = () => {
  const isCSR = useIsCSR();
  return (
    <div>
      {isCSR
        ? 'running in browser - client-side'
        : 'not running in browser - server side'}
    </div>
  );
};
```
